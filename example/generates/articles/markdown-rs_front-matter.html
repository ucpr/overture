const body = "+++
title = \"Test Title\"
tags = [\"rust\", \"test\"]
+++
"

let tree = markdown::to_mdast(body, config).ok().unwrap();

HugaHuga

tree.children().into_iter().for_each(|node| {
    for child in node.iter() {
        match child {
            markdown::mdast::Node::Toml(toml) => {
                front_matter = toml.value.clone();
            }
            _ => {}
        }
    }
});
